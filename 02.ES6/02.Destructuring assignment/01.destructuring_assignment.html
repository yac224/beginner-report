function parameter

      let test = function({ a = 'test1', b = 'test2' } = {}) {
        console.log('a:', a)
        console.log('b:', b)
      }
      console.log('=== a & b ===')
      test({ a: 1, b: 2 })

      console.log('=== only a ===')
      test({ a: 1 })

      console.log('=== only b ===')
      test({ b: 2 })

      console.log('=== nothing ===')
      test()
